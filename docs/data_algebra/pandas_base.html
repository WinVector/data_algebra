  25def none_mark_scalar_or_length(v) -> Optional[int]:
  26    """
  27    Test if item is a scalar (returning None) if it is, else length of object.
  28
  29    :param v: value to test
  30    :return: None if value is a scalar, else length.
  31    """
  32    # get some of the obvious types, and str (as str doesn't throw on len)
  33    if isinstance(v, (type(None), str, int, float)):
  34        return None  # obvious scalar
  35    # len() throws on scalars other than str
  36    try:
  37        return len(v)
  38    except TypeError:
  39        return None  # len() failed, probably a scalar

  42def promote_scalar_to_array(vi, *, target_len: int) -> List:
  43    """
  44    Convert a scalar into a vector. Pass a non-trivial array through.
  45
  46    :param vi: value to promote to scalar
  47    :target_len: length for vector
  48    :return: list
  49    """
  50    assert isinstance(target_len, int)
  51    assert target_len >= 0
  52    if target_len <= 0:
  53        return []
  54    len_v = none_mark_scalar_or_length(vi)
  55    # noinspection PyBroadException
  56    if len_v is None:
  57        return [vi] * target_len  # scalar
  58    assert len_v == target_len
  59    return vi

  70def _type_safe_equal(a, b):
  71    # Could try numpy.logical_and(numpy.greater_equal(a, b), numpy.less_equal(a, b)).
  72    # However, None >= None throws
  73    type_a = data_algebra.util.guess_carried_scalar_type(a)
  74    type_b = data_algebra.util.guess_carried_scalar_type(b)
  75    if not data_algebra.util.compatible_types([type_a, type_b]):
  76        raise TypeError(f"can't compare {type_a} to {type_b}")
  77    return numpy.equal(a, b)

  81    # Could try numpy.logical_or(numpy.greater(a, b), numpy.less(a, b)).
  82    # However, None > None throws

  90def _type_safe_is_in(a, b):
  91    b = list(b)
  92    if len(b) > 0:
  93        type_a = data_algebra.util.guess_carried_scalar_type(a)
  94        type_b = {data_algebra.util.map_type_to_canonical(type(v)) for v in b}
  95        if not data_algebra.util.compatible_types(type_b):
  96            raise TypeError(f"multiple types in set: {type_b}")
  97        type_b = list(type_b)[0]
  98        if not data_algebra.util.compatible_types([type_a, type_b]):
  99            raise TypeError(f"can't check for an {type_a} in a set of {type_b}'s")
 100    return numpy.isin(a, b)

 181    def _calc_date_diff(self, x0, x1):
 182        # x is a pandas Series or list of datetime.date compatible types
 183        x0 = self.pd.Series(x0)
 184        x1 = self.pd.Series(x1)
 185        x0_dates = self.pd.to_datetime(x0).dt.date.copy()
 186        x1_dates = self.pd.to_datetime(x1).dt.date.copy()
 187        deltas = [(x0_dates[i] - x1_dates[i]).days for i in range(len(x0_dates))]
 188        return deltas

 190    def _calc_base_Sunday(self, x):
 191        # x is a pandas Series or list of datetime.date compatible types
 192        x = self.pd.Series(x)
 193        x_dates = self.pd.to_datetime(x).dt.date.copy()
 194        res = [xi - datetime.timedelta(days=(xi.weekday() + 1) % 7) for xi in x_dates]
 195        return res

 197    def _calc_week_of_Year(self, x):
 198        # x is a pandas Series or list of datetime.date compatible types
 199        # TODO: better impl
 200        # Note was getting inconsistent results on vectorized methods
 201        x = self.pd.Series(x)

 211    def _coalesce(self, a, b):
 212        a_is_series = isinstance(a, self.pd.Series)
 213        b_is_series = isinstance(b, self.pd.Series)
 214        if (not a_is_series) and (not b_is_series):
 215            raise ValueError("at least one argument must be a Pandas series")
 216        if not a_is_series:
 217            a = self.pd.Series(numpy.array([a] * len(b)))
 218        if not b_is_series:
 219            b = self.pd.Series(numpy.array([b] * len(a)))
 220        res = a.combine_first(b)
 221        return res

 223    def _map_v(self, a, value_map, default_value=None):
 224        """Map values to values."""
 225        if len(value_map) > 0:
 226            type_a = data_algebra.util.guess_carried_scalar_type(a)
 227            type_k = {
 228                data_algebra.util.map_type_to_canonical(type(v)) for v in value_map.keys()
 229            }
 230            if not data_algebra.util.compatible_types(type_k):
 231                raise TypeError(f"multiple types in dictionary keys: {type_k} in mapv()")
 232            type_k = list(type_k)[0]
 233            if not data_algebra.util.compatible_types([type_a, type_k]):
 234                raise TypeError(f"can't map {type_a} from a dict of {type_k}'s in mapv()")
 235            type_v = {
 236                data_algebra.util.map_type_to_canonical(type(v)) for v in value_map.values()
 237            }
 238            if not data_algebra.util.compatible_types(type_v):
 239                raise TypeError(f"multiple types in dictionary values: {type_v} in mapv()")
 240            type_v = list(type_v)[0]
 241            if default_value is not None:
 242                type_d = data_algebra.util.guess_carried_scalar_type(default_value)
 243                if not data_algebra.util.compatible_types([type_d, type_v]):
 244                    raise TypeError(f"default is {type_d} for {type_v} values in mapv()'s")
 245        # https://pandas.pydata.org/docs/reference/api/pandas.Series.map.html
 246        a = self.pd.Series(a)
 247        a = a.map(value_map, na_action="ignore")
 248        a[self.bad_column_positions(a)] = default_value
 249        return numpy.array(a.values)

 251    def _if_else_expr(self, *args):
 252        """
 253        if_else(cond, a, b) returns a for positions where cond is True, b when cond is False, None otherwise.
 254        """
 255        assert len(args) == 3
 256        cond = args[0]
 257        a = args[1]
 258        b = args[2]
 259        res = numpy.where(cond, a, b)
 260        bad_posns = self.bad_column_positions(cond)
 261        if numpy.any(bad_posns):
 262            res[bad_posns] = None
 263        return res

 449    def _table_step(self, op, *, data_map: dict):
 450        """
 451        Return a copy of data frame from table description and data_map.
 452        """
 453        if op.node_name != "TableDescription":
 454            raise TypeError(
 455                "op was supposed to be a data_algebra.data_ops.TableDescription"
 456            )
 457        if (data_map is not None) and (len(data_map) > 0):
 458            df = data_map[op.table_name]
 459            if not self.is_appropriate_data_instance(df):
 460                raise ValueError(
 461                    "data_map[" + op.table_name + "] was not the right type"
 462                )
 463        else:
 464            df = op.head
 465            assert df is not None
 466            if not self.is_appropriate_data_instance(df):
 467                raise ValueError(
 468                    "Unnamed TableDescription stored was not the right type"
 469                )
 470        # check all columns we expect are present
 471        columns_using = op.column_names
 472        missing = set(columns_using) - set(df.columns)
 473        if len(missing) > 0:
 474            raise ValueError("missing required columns: " + str(missing))
 475        # make an index-free copy of the data to isolate side-effects and not deal with indices
 476        res = df.loc[:, columns_using]
 477        res = self.clean_copy(res)
 478        return res

 489    def columns_to_frame_(self, cols: Dict[str, Any], *, target_rows: Optional[int] = None):
 490        """
 491        Convert a dictionary of column names to series-like objects and scalars into a Pandas data frame.
 492        Deal with special cases, such as some columns coming in as scalars (often from Panda aggregation).
 493
 494        :param cols: dictionary mapping column names to columns
 495        :param target_rows: number of rows we are shooting for
 496        :return: Pandas data frame.
 497        """
 498        # noinspection PyUnresolvedReferences
 499        assert isinstance(cols, dict)
 500        assert isinstance(target_rows, (int, type(None)))
 501        if target_rows is not None:
 502            assert target_rows >= 0
 503        if len(cols) < 1:
 504            # all scalars, so nothing carrying index information
 505            if target_rows is not None:
 506                res = self.pd.DataFrame({}, index=range(target_rows))
 507                self.drop_indices(res)
 508                return res
 509            else:
 510                return self.pd.DataFrame({})
 511        was_all_scalars = True
 512        for v in cols.values():
 513            ln = none_mark_scalar_or_length(v)
 514            if ln is not None:
 515                was_all_scalars = False
 516                if target_rows is None:
 517                    target_rows = ln
 518                else:
 519                    assert target_rows == ln
 520        if was_all_scalars:
 521            if target_rows is None:
 522                target_rows = 1
 523            # all scalars, so nothing carrying index information
 524            promoted_cols = {k: promote_scalar_to_array(v, target_len=target_rows) for (k, v) in cols.items()}
 525            return self.clean_copy(self.pd.DataFrame(promoted_cols, index=range(target_rows)))
 526        assert target_rows is not None
 527        if target_rows < 1:
 528            # no rows, so presuming no index information (shouldn't have come from an aggregation)
 529            return self.pd.DataFrame({k: [] for k in cols.keys()})
 530        # agg can return scalars, which then can't be made into a self.pd.DataFrame
 531        promoted_cols = {k: promote_scalar_to_array(v, target_len=target_rows) for (k, v) in cols.items()}
 532        return self.pd.DataFrame(promoted_cols)

 534    def add_data_frame_columns_to_data_frame_(self, res, transient_new_frame):
 535        """
 536        Add columns from transient_new_frame to res. Res may be altered, and either of res or
 537        transient_new_frame may be returned.
 538        """
 539        if transient_new_frame.shape[1] < 1:
 540            return res
 541        if (res.shape[0] == 0) and (transient_new_frame.shape[0] > 0):
 542            # scalars get interpreted as single row items, instead of zero row items
 543            # growing the extension frame
 544            transient_new_frame = self.clean_copy(transient_new_frame.iloc[range(0), :])
 545        if res.shape[0] == transient_new_frame.shape[0]:
 546            if res.shape[1] < 1:
 547                return transient_new_frame
 548            if transient_new_frame.shape[1] < 1:
 549                return res
 550        if (2 * transient_new_frame.shape[1]) > res.shape[1]:
 551            # lots of columns path
 552            # https://win-vector.com/2021/08/03/i-think-pandas-may-have-lost-the-plot/
 553            for c in set(res.columns).intersection(set(transient_new_frame.columns)):
 554                del res[c]
 555            return self.pd.concat([res, transient_new_frame], axis=1)
 556        # normal path
 557        for c in transient_new_frame.columns:
 558            res[c] = transient_new_frame[c]
 559        return res

 581    def _extend_step(self, op, *, data_map):
 582        """
 583        Execute an extend step, returning a data frame.
 584        """
 585        if op.node_name != "ExtendNode":
 586            raise TypeError("op was supposed to be a data_algebra.data_ops.ExtendNode")
 587        res = self._eval_value_source(op.sources[0], data_map=data_map)
 588        if res.shape[0] <= 0:
 589            # special case out no-row frame
 590            incoming_col_set = set(res.columns)
 591            v_dict = {k: [] for k in res.columns}
 592            for k in op.ops.keys():
 593                if k not in incoming_col_set:
 594                    v_dict[k] = []
 595            return self.pd.DataFrame(v_dict)
 596        window_situation = (
 597            op.windowed_situation
 598            or (len(op.partition_by) > 0)
 599            or (len(op.order_by) > 0)
 600        )
 601        if window_situation:
 602            op.check_extend_window_fns_()
 603        if not window_situation:
 604            with warnings.catch_warnings():
 605                warnings.simplefilter(
 606                    "ignore"
 607                )  # out of range things like arccosh were warning
 608                new_cols = {k: opk.act_on(res, data_model=self) for k, opk in op.ops.items()}
 609            new_frame = self.columns_to_frame_(new_cols, target_rows=res.shape[0])
 610            res = self.add_data_frame_columns_to_data_frame_(res, new_frame)
 611        else:
 612            data_algebra_temp_cols = {}
 613            standin_name = "_data_algebra_temp_g"  # name of an arbitrary input variable
 614            # build up a sub-frame to work on
 615            col_list = [c for c in set(op.partition_by)]
 616            col_set = set(col_list)
 617            for c in op.order_by:
 618                if c not in col_set:
 619                    col_list.append(c)
 620                    col_set.add(c)
 621            order_cols = [c for c in col_list]  # must be partition by followed by order
 622            for (k, opk) in op.ops.items():
 623                # assumes all args are column names or values, enforce this earlier
 624                if len(opk.args) > 0:
 625                    if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 626                        value_name = opk.args[0].column_name
 627                        if value_name not in col_set:
 628                            col_list.append(value_name)
 629                            col_set.add(value_name)
 630                    elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 631                        key = str(opk.args[0].value)
 632                        if key not in data_algebra_temp_cols.keys():
 633                            value_name = "data_algebra_extend_temp_col_" + str(
 634                                len(data_algebra_temp_cols)
 635                            )
 636                            data_algebra_temp_cols[key] = value_name
 637                            col_list.append(value_name)
 638                            res[value_name] = opk.args[0].value
 639                    else:
 640                        raise ValueError("opk must be a ColumnReference or Value")
 641            ascending = [c not in set(op.reverse) for c in col_list]
 642            subframe = self.clean_copy(res[col_list])
 643            subframe["_data_algebra_orig_index"] = subframe.index
 644            if len(order_cols) > 0:
 645                subframe = self.clean_copy(subframe.sort_values(
 646                    by=col_list, ascending=ascending
 647                ))
 648            subframe[standin_name] = 1
 649            if len(op.partition_by) > 0:
 650                opframe = subframe.groupby(op.partition_by, observed=True)
 651                #  Groupby preserves the order of rows within each group.
 652                # https://pandas.pydata.org/pandas-docs/stable/reference/api/pandas.DataFrame.groupby.html
 653            else:
 654                opframe = subframe.groupby([standin_name], observed=True)
 655            # perform calculations
 656            for (k, opk) in op.ops.items():
 657                # work on a slice of the data frame
 658                # Availability roughly documented in:
 659                # https://github.com/WinVector/data_algebra/blob/main/Examples/Methods/data_algebra_catalog.ipynb
 660                # (essentially none but _ngroup)
 661                if len(opk.args) <= 0:
 662                    # check for and remove initial underbar
 663                    assert isinstance(opk.op, str)
 664                    assert len(opk.op) > 1
 665                    assert opk.op[0] == "_"
 666                    zero_op = opk.op[1:]
 667                    if zero_op in {"row_number", "count"}:
 668                        subframe[k] = opframe.cumcount() + 1
 669                    elif zero_op in {"ngroup"}:
 670                        subframe[k] = opframe.ngroup()
 671                    elif zero_op in {"size"}:
 672                        transform_op = zero_op
 673                        try:
 674                            transform_op = self.transform_op_map[transform_op]
 675                        except KeyError:
 676                            pass
 677                        subframe[k] = opframe[standin_name].transform(
 678                            transform_op
 679                        )  # Pandas transform, not data_algebra
 680                    else:
 681                        raise KeyError(
 682                            "not implemented in windowed situation: "
 683                            + str(k)
 684                            + ": "
 685                            + str(opk)
 686                        )
 687                else:
 688                    transform_args = []
 689                    if len(opk.args) > 1:
 690                        for i in range(1, len(opk.args)):
 691                            assert isinstance(opk.args[i], data_algebra.expr_rep.Value)
 692                        transform_args = [
 693                            opk.args[i].value for i in range(1, len(opk.args))
 694                        ]
 695                    if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 696                        value_name = opk.args[0].column_name
 697                        if value_name not in set(col_list):
 698                            col_list.append(value_name)
 699                        transform_op = opk.op
 700                        try:
 701                            transform_op = self.transform_op_map[transform_op]
 702                        except KeyError:
 703                            pass
 704                        subframe[k] = opframe[value_name].transform(
 705                            transform_op, *transform_args
 706                        )  # Pandas transform, not data_algegra
 707                    elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 708                        value_name = data_algebra_temp_cols[str(opk.args[0].value)]
 709                        transform_op = opk.op
 710                        try:
 711                            transform_op = self.transform_op_map[transform_op]
 712                        except KeyError:
 713                            pass
 714                        subframe[k] = opframe[value_name].transform(
 715                            transform_op, *transform_args
 716                        )  # Pandas transform, not data_algegra
 717                    else:
 718                        raise ValueError(
 719                            f"opk must be a ColumnReference or Value ({opk})"
 720                        )
 721            # clear some temps
 722            for value_name in data_algebra_temp_cols.values():
 723                del res[value_name]
 724            # copy out results
 725            subframe = subframe.sort_values(by=["_data_algebra_orig_index"])
 726            subframe = subframe.loc[:, list(op.ops.keys())]
 727            subframe = self.clean_copy(subframe)
 728            res = self.add_data_frame_columns_to_data_frame_(res, subframe)
 729        return res

 731    def _project_step(self, op, *, data_map):
 732        """
 733        Execute a project step, returning a data frame.
 734        """
 735        if op.node_name != "ProjectNode":
 736            raise TypeError("op was supposed to be a data_algebra.data_ops.ProjectNode")
 737        # check these are forms we are prepared to work with, and build an aggregation dictionary
 738        # build an agg list: https://www.shanelynn.ie/summarising-aggregation-and-grouping-data-in-python-pandas/
 739        # https://stackoverflow.com/questions/44635626/rename-result-columns-from-pandas-aggregation-futurewarning-using-a-dict-with
 740        # try the following tutorial:
 741        # https://www.shanelynn.ie/summarising-aggregation-and-grouping-data-in-python-pandas/
 742        data_algebra_temp_cols = {}
 743        res = self._eval_value_source(op.sources[0], data_map=data_map)
 744        for (k, opk) in op.ops.items():
 745            if len(opk.args) > 1:
 746                raise ValueError(
 747                    "non-trivial aggregation expression: " + str(k) + ": " + str(opk)
 748                )
 749            if len(opk.args) > 0:
 750                if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 751                    pass
 752                elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 753                    key = str(opk.args[0].value)
 754                    if key not in data_algebra_temp_cols.keys():
 755                        value_name = "data_algebra_project_temp_col_" + str(
 756                            len(data_algebra_temp_cols)
 757                        )
 758                        data_algebra_temp_cols[key] = value_name
 759                        res[value_name] = opk.args[0].value
 760                else:
 761                    raise ValueError(
 762                        "windows expression argument must be a column or value: "
 763                        + str(k)
 764                        + ": "
 765                        + str(opk)
 766                    )
 767        res["_data_table_temp_col"] = 1
 768        if len(op.group_by) > 0:
 769            res = res.groupby(op.group_by, observed=True)
 770        if len(op.ops) > 0:
 771            cols = {}
 772            for k, opk in op.ops.items():
 773                value_name = None
 774                if len(opk.args) > 0:
 775                    value_name = str(opk.args[0])
 776                    if isinstance(opk.args[0], data_algebra.expr_rep.Value):
 777                        value_name = data_algebra_temp_cols[value_name]
 778                transform_op = opk.op
 779                if len(opk.args) > 0:
 780                    transform_op = opk.op
 781                    try:
 782                        transform_op = self.transform_op_map[transform_op]
 783                    except KeyError:
 784                        pass
 785                    vk = res[value_name].agg(transform_op)
 786                else:
 787                    # expect and strip off initial underbar
 788                    assert isinstance(transform_op, str)
 789                    assert len(transform_op) > 1
 790                    assert transform_op[0] == "_"
 791                    transform_op = transform_op[1:]
 792                    try:
 793                        transform_op = self.transform_op_map[transform_op]
 794                    except KeyError:
 795                        pass
 796                    vk = res["_data_table_temp_col"].agg(transform_op)
 797                cols[k] = vk
 798        else:
 799            cols = {"_data_table_temp_col": res["_data_table_temp_col"].agg("sum")}
 800        # agg can return scalars, which then can't be made into a self.pd.DataFrame
 801        res = self.columns_to_frame_(cols)
 802        res = res.reset_index(
 803            drop=(len(op.group_by) < 1) or (res.shape[0] <= 0),
 804            inplace=False,
 805        )  # grouping variables in the index
 806        missing_group_cols = set(op.group_by) - set(res.columns)
 807        if res.shape[0] > 0:
 808            if len(missing_group_cols) != 0:
 809                raise ValueError("Missing column groups")
 810        else:
 811            for g in missing_group_cols:
 812                res[g] = []
 813        if "_data_table_temp_col" in res.columns:
 814            res = res.drop("_data_table_temp_col", axis=1, inplace=False)
 815        # double check shape is what we expect
 816        if not data_algebra.util.table_is_keyed_by_columns(res, op.group_by):
 817            raise ValueError("result wasn't keyed by group_by columns")
 818        return res

 920    def _natural_join_step(self, op, *, data_map):
 921        """
 922        Execute a natural join step, returning a data frame.
 923        """
 924        if op.node_name != "NaturalJoinNode":
 925            raise TypeError(
 926                "op was supposed to be a data_algebra.data_ops.NaturalJoinNode"
 927            )
 928        left = self._eval_value_source(op.sources[0], data_map=data_map)
 929        right = self._eval_value_source(op.sources[1], data_map=data_map)
 930        if (left.shape[0] == 0) and (right.shape[0] == 0):
 931            # pandas seems to not like this case
 932            return self.pd.DataFrame({k: [] for k in op.columns_produced()})
 933        common_cols = set([c for c in left.columns]).intersection(
 934            [c for c in right.columns]
 935        )
 936        type_checks = data_algebra.util.check_columns_appear_compatible(
 937            left, right, columns=common_cols
 938        )
 939        if type_checks is not None:
 940            raise ValueError(f"join: incompatible column types: {type_checks}")
 941        on_a = op.on_a
 942        on_b = op.on_b
 943        scratch_col = None  # extra column to prevent empty-on issues
 944        if len(on_a) <= 0:
 945            scratch_col = "data_algebra_temp_merge_col"
 946            on_a = [scratch_col]
 947            on_b = [scratch_col]
 948            left[scratch_col] = 1
 949            right[scratch_col] = 1
 950        # noinspection PyUnresolvedReferences
 951        res = self.pd.merge(
 952            left=left,
 953            right=right,
 954            how=self.standardize_join_code_(op.jointype),
 955            left_on=on_a,
 956            right_on=on_b,
 957            sort=False,
 958            suffixes=("", "_tmp_right_col"),
 959        )
 960        self.drop_indices(res)
 961        if scratch_col is not None:
 962            del res[scratch_col]
 963        on_a_set = set(op.on_a)
 964        for c in common_cols:
 965            if c not in on_a_set:
 966                is_null = res[c].isnull()
 967                res.loc[is_null, c] = res.loc[is_null, c + "_tmp_right_col"]
 968                res = res.drop(c + "_tmp_right_col", axis=1, inplace=False)
 969        self.drop_indices(res)
 970        return res

1016    def blocks_to_rowrecs(self, data, *, blocks_in: data_algebra.cdata.RecordSpecification):
1017        """
1018        Convert a block record (record spanning multiple rows) into a rowrecord (record in a single row).
1019
1020        :param data: data frame to be transformed
1021        :param blocks_in: record specification
1022        :return: transformed data frame
1023        """
1024        assert isinstance(blocks_in, data_algebra.cdata.RecordSpecification)
1025        ck = [k for k in blocks_in.content_keys if k is not None]
1026        if len(ck) != len(set(ck)):
1027            raise ValueError("blocks_in can not have duplicate content keys")
1028        data = self.clean_copy(data)
1029        missing_cols = set(blocks_in.control_table_keys).union(blocks_in.record_keys) - set(
1030            data.columns
1031        )
1032        if len(missing_cols) > 0:
1033            raise KeyError("missing required columns: " + str(missing_cols))
1034        # table must be keyed by record_keys + control_table_keys
1035        if not data_algebra.util.table_is_keyed_by_columns(
1036            data, blocks_in.record_keys + blocks_in.control_table_keys
1037        ):
1038            raise ValueError(
1039                "table is not keyed by blocks_in.record_keys + blocks_in.control_table_keys"
1040            )
1041        # convert to row-records
1042        # regularize/complete records
1043        dtemp = data.copy()  # TODO: select down columns
1044        dtemp["FALSE_AGG_KEY"] = 1
1045        if len(blocks_in.record_keys) > 0:
1046            ideal = dtemp[blocks_in.record_keys + ["FALSE_AGG_KEY"]].copy()
1047            res = ideal.groupby(blocks_in.record_keys, observed=True)["FALSE_AGG_KEY"].agg("sum")
1048            ideal = self.data_frame(res).reset_index(drop=False, inplace=False)
1049            ideal["FALSE_AGG_KEY"] = 1
1050            ctemp = blocks_in.control_table[blocks_in.control_table_keys].copy()
1051            ctemp["FALSE_AGG_KEY"] = 1
1052            ideal = ideal.merge(ctemp, how="outer", on="FALSE_AGG_KEY")
1053            self.drop_indices(ideal)
1054            dtemp = ideal.merge(
1055                right=dtemp,
1056                how="left",
1057                on=blocks_in.record_keys + blocks_in.control_table_keys + ["FALSE_AGG_KEY"],
1058            )
1059        dtemp = dtemp.sort_values(
1060            by=blocks_in.record_keys + blocks_in.control_table_keys, inplace=False, ignore_index=True
1061        )
1062        self.drop_indices(dtemp)
1063        # start building up result frame
1064        if len(blocks_in.record_keys) > 0:
1065            res = dtemp.groupby(blocks_in.record_keys, observed=True)["FALSE_AGG_KEY"].agg("sum")
1066        else:
1067            res = dtemp.groupby("FALSE_AGG_KEY", observed=True)["FALSE_AGG_KEY"].agg("sum")
1068        res = self.data_frame(res).reset_index(drop=False, inplace=False)
1069        res = res.sort_values(by=blocks_in.record_keys, inplace=False, ignore_index=True)
1070        self.drop_indices(res)
1071        del res["FALSE_AGG_KEY"]
1072        # now fill in columns
1073        ckeys = blocks_in.control_table_keys
1074        value_keys = [k for k in blocks_in.control_table.columns if k not in set(ckeys)]
1075        donor_cols = set(dtemp.columns)
1076        for i in range(blocks_in.control_table.shape[0]):
1077            want = numpy.ones((dtemp.shape[0],), dtype=bool)
1078            for ck in ckeys:
1079                want = numpy.logical_and(want, dtemp[ck] == blocks_in.control_table[ck][i])
1080            if numpy.any(want):
1081                for vk in value_keys:
1082                    if vk in donor_cols:
1083                        dcol = blocks_in.control_table[vk][i]
1084                        res[dcol] = numpy.asarray(dtemp.loc[want, vk])
1085        # fill in any missed columns
1086        colset = set(res.columns)
1087        for c in blocks_in.row_version():
1088            if c not in colset:
1089                res[c] = None
1090        if data.shape[0] <= 0:
1091            res = res.loc[range(0), :]
1092            res = self.clean_copy(res)
1093        return res

 541            return res
 542        if (res.shape[0] == 0) and (transient_new_frame.shape[0] > 0):
 543            # scalars get interpreted as single row items, instead of zero row items
 544            # growing the extension frame
 545            transient_new_frame = self.clean_copy(transient_new_frame.iloc[range(0), :])
 546        if res.shape[0] == transient_new_frame.shape[0]:
 547            if res.shape[1] < 1:
 548                return transient_new_frame
 549            if transient_new_frame.shape[1] < 1:
 550                return res
 551        if (2 * transient_new_frame.shape[1]) > res.shape[1]:
 552            # lots of columns path
 553            # https://win-vector.com/2021/08/03/i-think-pandas-may-have-lost-the-plot/
 554            for c in set(res.columns).intersection(set(transient_new_frame.columns)):
 555                del res[c]
 556            return self.pd.concat([res, transient_new_frame], axis=1)
 557        # normal path
 558        for c in transient_new_frame.columns:
 559            res[c] = transient_new_frame[c]
 560        return res

 582    def _extend_step(self, op, *, data_map):
 583        """
 584        Execute an extend step, returning a data frame.
 585        """
 586        if op.node_name != "ExtendNode":
 587            raise TypeError("op was supposed to be a data_algebra.data_ops.ExtendNode")
 588        res = self._eval_value_source(op.sources[0], data_map=data_map)
 589        if res.shape[0] <= 0:
 590            # special case out no-row frame
 591            incoming_col_set = set(res.columns)
 592            v_dict = {k: [] for k in res.columns}
 593            for k in op.ops.keys():
 594                if k not in incoming_col_set:
 595                    v_dict[k] = []
 596            return self.pd.DataFrame(v_dict)
 597        window_situation = (
 598            op.windowed_situation
 599            or (len(op.partition_by) > 0)
 600            or (len(op.order_by) > 0)
 601        )
 602        if window_situation:
 603            op.check_extend_window_fns_()
 604        if not window_situation:
 605            with warnings.catch_warnings():
 606                warnings.simplefilter(
 607                    "ignore"
 608                )  # out of range things like arccosh were warning
 609                new_cols = {k: opk.act_on(res, data_model=self) for k, opk in op.ops.items()}
 610            new_frame = self.columns_to_frame_(new_cols, target_rows=res.shape[0])
 611            res = self.add_data_frame_columns_to_data_frame_(res, new_frame)
 612        else:
 613            data_algebra_temp_cols = {}
 614            standin_name = "_data_algebra_temp_g"  # name of an arbitrary input variable
 615            # build up a sub-frame to work on
 616            col_list = [c for c in set(op.partition_by)]
 617            col_set = set(col_list)
 618            for c in op.order_by:
 619                if c not in col_set:
 620                    col_list.append(c)
 621                    col_set.add(c)
 622            order_cols = [c for c in col_list]  # must be partition by followed by order
 623            for (k, opk) in op.ops.items():
 624                # assumes all args are column names or values, enforce this earlier
 625                if len(opk.args) > 0:
 626                    if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 627                        value_name = opk.args[0].column_name
 628                        if value_name not in col_set:
 629                            col_list.append(value_name)
 630                            col_set.add(value_name)
 631                    elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 632                        key = str(opk.args[0].value)
 633                        if key not in data_algebra_temp_cols.keys():
 634                            value_name = "data_algebra_extend_temp_col_" + str(
 635                                len(data_algebra_temp_cols)
 636                            )
 637                            data_algebra_temp_cols[key] = value_name
 638                            col_list.append(value_name)
 639                            res[value_name] = opk.args[0].value
 640                    else:
 641                        raise ValueError("opk must be a ColumnReference or Value")
 642            ascending = [c not in set(op.reverse) for c in col_list]
 643            subframe = self.clean_copy(res[col_list])
 644            subframe["_data_algebra_orig_index"] = subframe.index
 645            if len(order_cols) > 0:
 646                subframe = self.clean_copy(subframe.sort_values(
 647                    by=col_list, ascending=ascending
 648                ))
 649            subframe[standin_name] = 1
 650            if len(op.partition_by) > 0:
 651                opframe = subframe.groupby(op.partition_by, observed=True)
 652                #  Groupby preserves the order of rows within each group.
 653                # https://pandas.pydata.org/pandas-docs/stable/reference/api/pandas.DataFrame.groupby.html
 654            else:
 655                opframe = subframe.groupby([standin_name], observed=True)
 656            # perform calculations
 657            for (k, opk) in op.ops.items():
 658                # work on a slice of the data frame
 659                # Availability roughly documented in:
 660                # https://github.com/WinVector/data_algebra/blob/main/Examples/Methods/data_algebra_catalog.ipynb
 661                # (essentially none but _ngroup)
 662                if len(opk.args) <= 0:
 663                    # check for and remove initial underbar
 664                    assert isinstance(opk.op, str)
 665                    assert len(opk.op) > 1
 666                    assert opk.op[0] == "_"
 667                    zero_op = opk.op[1:]
 668                    if zero_op in {"row_number", "count"}:
 669                        subframe[k] = opframe.cumcount() + 1
 670                    elif zero_op in {"ngroup"}:
 671                        subframe[k] = opframe.ngroup()
 672                    elif zero_op in {"size"}:
 673                        transform_op = zero_op
 674                        try:
 675                            transform_op = self.transform_op_map[transform_op]
 676                        except KeyError:
 677                            pass
 678                        subframe[k] = opframe[standin_name].transform(
 679                            transform_op
 680                        )  # Pandas transform, not data_algebra
 681                    else:
 682                        raise KeyError(
 683                            "not implemented in windowed situation: "
 684                            + str(k)
 685                            + ": "
 686                            + str(opk)
 687                        )
 688                else:
 689                    transform_args = []
 690                    if len(opk.args) > 1:
 691                        for i in range(1, len(opk.args)):
 692                            assert isinstance(opk.args[i], data_algebra.expr_rep.Value)
 693                        transform_args = [
 694                            opk.args[i].value for i in range(1, len(opk.args))
 695                        ]
 696                    if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 697                        value_name = opk.args[0].column_name
 698                        if value_name not in set(col_list):
 699                            col_list.append(value_name)
 700                        transform_op = opk.op
 701                        try:
 702                            transform_op = self.transform_op_map[transform_op]
 703                        except KeyError:
 704                            pass
 705                        subframe[k] = opframe[value_name].transform(
 706                            transform_op, *transform_args
 707                        )  # Pandas transform, not data_algegra
 708                    elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 709                        value_name = data_algebra_temp_cols[str(opk.args[0].value)]
 710                        transform_op = opk.op
 711                        try:
 712                            transform_op = self.transform_op_map[transform_op]
 713                        except KeyError:
 714                            pass
 715                        subframe[k] = opframe[value_name].transform(
 716                            transform_op, *transform_args
 717                        )  # Pandas transform, not data_algegra
 718                    else:
 719                        raise ValueError(
 720                            f"opk must be a ColumnReference or Value ({opk})"
 721                        )
 722            # clear some temps
 723            for value_name in data_algebra_temp_cols.values():
 724                del res[value_name]
 725            # copy out results
 726            subframe = subframe.sort_values(by=["_data_algebra_orig_index"])
 727            subframe = subframe.loc[:, list(op.ops.keys())]
 728            subframe = self.clean_copy(subframe)
 729            res = self.add_data_frame_columns_to_data_frame_(res, subframe)
 730        return res

 732    def _project_step(self, op, *, data_map):
 733        """
 734        Execute a project step, returning a data frame.
 735        """
 736        if op.node_name != "ProjectNode":
 737            raise TypeError("op was supposed to be a data_algebra.data_ops.ProjectNode")
 738        # check these are forms we are prepared to work with, and build an aggregation dictionary
 739        # build an agg list: https://www.shanelynn.ie/summarising-aggregation-and-grouping-data-in-python-pandas/
 740        # https://stackoverflow.com/questions/44635626/rename-result-columns-from-pandas-aggregation-futurewarning-using-a-dict-with
 741        # try the following tutorial:
 742        # https://www.shanelynn.ie/summarising-aggregation-and-grouping-data-in-python-pandas/
 743        data_algebra_temp_cols = {}
 744        res = self._eval_value_source(op.sources[0], data_map=data_map)
 745        for (k, opk) in op.ops.items():
 746            if len(opk.args) > 1:
 747                raise ValueError(
 748                    "non-trivial aggregation expression: " + str(k) + ": " + str(opk)
 749                )
 750            if len(opk.args) > 0:
 751                if isinstance(opk.args[0], data_algebra.expr_rep.ColumnReference):
 752                    pass
 753                elif isinstance(opk.args[0], data_algebra.expr_rep.Value):
 754                    key = str(opk.args[0].value)
 755                    if key not in data_algebra_temp_cols.keys():
 756                        value_name = "data_algebra_project_temp_col_" + str(
 757                            len(data_algebra_temp_cols)
 758                        )
 759                        data_algebra_temp_cols[key] = value_name
 760                        res[value_name] = opk.args[0].value
 761                else:
 762                    raise ValueError(
 763                        "windows expression argument must be a column or value: "
 764                        + str(k)
 765                        + ": "
 766                        + str(opk)
 767                    )
 768        res["_data_table_temp_col"] = 1
 769        if len(op.group_by) > 0:
 770            res = res.groupby(op.group_by, observed=True)
 771        if len(op.ops) > 0:
 772            cols = {}
 773            for k, opk in op.ops.items():
 774                value_name = None
 775                if len(opk.args) > 0:
 776                    value_name = str(opk.args[0])
 777                    if isinstance(opk.args[0], data_algebra.expr_rep.Value):
 778                        value_name = data_algebra_temp_cols[value_name]
 779                transform_op = opk.op
 780                if len(opk.args) > 0:
 781                    transform_op = opk.op
 782                    try:
 783                        transform_op = self.transform_op_map[transform_op]
 784                    except KeyError:
 785                        pass
 786                    vk = res[value_name].agg(transform_op)
 787                else:
 788                    # expect and strip off initial underbar
 789                    assert isinstance(transform_op, str)
 790                    assert len(transform_op) > 1
 791                    assert transform_op[0] == "_"
 792                    transform_op = transform_op[1:]
 793                    try:
 794                        transform_op = self.transform_op_map[transform_op]
 795                    except KeyError:
 796                        pass
 797                    vk = res["_data_table_temp_col"].agg(transform_op)
 798                cols[k] = vk
 799        else:
 800            cols = {"_data_table_temp_col": res["_data_table_temp_col"].agg("sum")}
 801        # agg can return scalars, which then can't be made into a self.pd.DataFrame
 802        res = self.columns_to_frame_(cols)
 803        res = res.reset_index(
 804            drop=(len(op.group_by) < 1) or (res.shape[0] <= 0),
 805            inplace=False,
 806        )  # grouping variables in the index
 807        missing_group_cols = set(op.group_by) - set(res.columns)
 808        if res.shape[0] > 0:
 809            if len(missing_group_cols) != 0:
 810                raise ValueError("Missing column groups")
 811        else:
 812            for g in missing_group_cols:
 813                res[g] = []
 814        if "_data_table_temp_col" in res.columns:
 815            res = res.drop("_data_table_temp_col", axis=1, inplace=False)
 816        # double check shape is what we expect
 817        if not data_algebra.util.table_is_keyed_by_columns(res, op.group_by):
 818            raise ValueError("result wasn't keyed by group_by columns")
 819        return res

 921    def _natural_join_step(self, op, *, data_map):
 922        """
 923        Execute a natural join step, returning a data frame.
 924        """
 925        if op.node_name != "NaturalJoinNode":
 926            raise TypeError(
 927                "op was supposed to be a data_algebra.data_ops.NaturalJoinNode"
 928            )
 929        left = self._eval_value_source(op.sources[0], data_map=data_map)
 930        right = self._eval_value_source(op.sources[1], data_map=data_map)
 931        if (left.shape[0] == 0) and (right.shape[0] == 0):
 932            # pandas seems to not like this case
 933            return self.pd.DataFrame({k: [] for k in op.columns_produced()})
 934        common_cols = set([c for c in left.columns]).intersection(
 935            [c for c in right.columns]
 936        )
 937        type_checks = data_algebra.util.check_columns_appear_compatible(
 938            left, right, columns=common_cols
 939        )
 940        if type_checks is not None:
 941            raise ValueError(f"join: incompatible column types: {type_checks}")
 942        on_a = op.on_a
 943        on_b = op.on_b
 944        scratch_col = None  # extra column to prevent empty-on issues
 945        if len(on_a) <= 0:
 946            scratch_col = "data_algebra_temp_merge_col"
 947            on_a = [scratch_col]
 948            on_b = [scratch_col]
 949            left[scratch_col] = 1
 950            right[scratch_col] = 1
 951        # noinspection PyUnresolvedReferences
 952        res = self.pd.merge(
 953            left=left,
 954            right=right,
 955            how=self.standardize_join_code_(op.jointype),
 956            left_on=on_a,
 957            right_on=on_b,
 958            sort=False,
 959            suffixes=("", "_tmp_right_col"),
 960        )
 961        self.drop_indices(res)
 962        if scratch_col is not None:
 963            del res[scratch_col]
 964        on_a_set = set(op.on_a)
 965        for c in common_cols:
 966            if c not in on_a_set:
 967                is_null = res[c].isnull()
 968                res.loc[is_null, c] = res.loc[is_null, c + "_tmp_right_col"]
 969                res = res.drop(c + "_tmp_right_col", axis=1, inplace=False)
 970        self.drop_indices(res)
 971        return res

1017    def blocks_to_rowrecs(self, data, *, blocks_in: data_algebra.cdata.RecordSpecification):
1018        """
1019        Convert a block record (record spanning multiple rows) into a rowrecord (record in a single row).
1020
1021        :param data: data frame to be transformed
1022        :param blocks_in: record specification
1023        :return: transformed data frame
1024        """
1025        assert isinstance(blocks_in, data_algebra.cdata.RecordSpecification)
1026        ck = [k for k in blocks_in.content_keys if k is not None]
1027        if len(ck) != len(set(ck)):
1028            raise ValueError("blocks_in can not have duplicate content keys")
1029        data = self.clean_copy(data)
1030        missing_cols = set(blocks_in.control_table_keys).union(blocks_in.record_keys) - set(
1031            data.columns
1032        )
1033        if len(missing_cols) > 0:
1034            raise KeyError("missing required columns: " + str(missing_cols))
1035        # table must be keyed by record_keys + control_table_keys
1036        if not data_algebra.util.table_is_keyed_by_columns(
1037            data, blocks_in.record_keys + blocks_in.control_table_keys
1038        ):
1039            raise ValueError(
1040                "table is not keyed by blocks_in.record_keys + blocks_in.control_table_keys"
1041            )
1042        # convert to row-records
1043        # regularize/complete records
1044        dtemp = data.copy()  # TODO: select down columns
1045        dtemp["FALSE_AGG_KEY"] = 1
1046        if len(blocks_in.record_keys) > 0:
1047            ideal = dtemp[blocks_in.record_keys + ["FALSE_AGG_KEY"]].copy()
1048            res = ideal.groupby(blocks_in.record_keys, observed=True)["FALSE_AGG_KEY"].agg("sum")
1049            ideal = self.data_frame(res).reset_index(drop=False, inplace=False)
1050            ideal["FALSE_AGG_KEY"] = 1
1051            ctemp = blocks_in.control_table[blocks_in.control_table_keys].copy()
1052            ctemp["FALSE_AGG_KEY"] = 1
1053            ideal = ideal.merge(ctemp, how="outer", on="FALSE_AGG_KEY")
1054            self.drop_indices(ideal)
1055            dtemp = ideal.merge(
1056                right=dtemp,
1057                how="left",
1058                on=blocks_in.record_keys + blocks_in.control_table_keys + ["FALSE_AGG_KEY"],
1059            )
1060        dtemp = dtemp.sort_values(
1061            by=blocks_in.record_keys + blocks_in.control_table_keys, inplace=False, ignore_index=True
1062        )
1063        self.drop_indices(dtemp)
1064        # start building up result frame
