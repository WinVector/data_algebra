119def _check_scalar_bad(x):
120    """
121    Return True if scalar value is none or nan, else False.
122    """
123
124    if x is None:
125        return True
126    if not isinstance(x, numbers.Number):
127        return False
128    if numpy.isinf(x) or numpy.isnan(x):
129        return True
130    return False
131
132
133def _check_scalar_nan(x):
134    """
135    Return True if scalar value is nan, else False.
136    """
137
138    if x is None:
139        return True  # sqlite can't tell this from nan
140    if not isinstance(x, numbers.Number):
141        return False
142    if numpy.isnan(x):
143        return True
144    return False

147def _check_scalar_inf(x):
148    """
149    Return True if scalar value is inf, else False.
150    """
151
152    if x is None:
153        return False
154    if not isinstance(x, numbers.Number):
155        return False
156    if numpy.isinf(x):
157        return True
158    return False

161def _sign_fn(x):
162    # noinspection PyBroadException
163    try:
164        if _check_scalar_bad(x):
165            return numpy.nan
166        if x > 0:
167            return 1.0
168        if x < 0:
169            return -1.0
170        return 0.0
171    except Exception:
172        return numpy.nan
173
174
175def _abs_fn(x):
176    if _check_scalar_bad(x):
177        return numpy.nan
178    if x >= 0:
179        return x
180    return -x

183def _wrap_scalar_fn(f, x):
184    if _check_scalar_bad(x):
185        return numpy.nan
186    with warnings.catch_warnings():
187        warnings.simplefilter("ignore")
188        return f(x)

207class CollectingAgg(abc.ABC):
208    """
209    Aggregate from a collection. SQLite user class.
210    """
211
212    def __init__(self):
213        self.collection = []
214
215    def step(self, value):
216        """
217        Observe value
218        """
219        if not _check_scalar_bad(value):
220            self.collection.append(value)
221
222    @abc.abstractmethod
223    def calc(self) -> float:
224        """
225        Perform the calculation (only called with non-trivial data)
226        """
227
228    def finalize(self):
229        """
230        Return result.
231        """
232        if len(self.collection) < 1:
233            res = numpy.nan
234        else:
235            res = self.calc()
236        self.collection = []
237        return res

253class SampVarDevAgg(CollectingAgg):
254    """
255    Aggregate as sample standard deviation. SQLite user class.
256    This version keeps the data instead of using the E[(x-E[x])^2] = E[x^2] - E[x]^2 formula
257    """
258
259    def __init__(self):
260        CollectingAgg.__init__(self)
261
262    def calc(self) -> float:
263        "do it"
264        # pandas.DataFrame({'x': [1., 3.], 'g': ['a', 'a']}).groupby(['g']).transform('var')['x']
265        # [2, 2]  # sample variance, what we want
266        # numpy.var([[1., 3.]])
267        # 1.0  # population variance- not what we want
268        n = len(self.collection)
269        if n < 2:
270            return numpy.nan
271        return float(numpy.var(self.collection) * n / (n - 1))

274class SampStdDevAgg(CollectingAgg):
275    """
276    Aggregate as sample standard deviation. SQLite user class.
277    This version keeps the data instead of using the E[(x-E[x])^2] = E[x^2] - E[x]^2 formula
278    """
279
280    def __init__(self):
281        CollectingAgg.__init__(self)
282
283    def calc(self) -> float:
284        "do it"
285        # pandas.DataFrame({'x': [1., 3.], 'g': ['a', 'a']}).groupby(['g']).transform('std')['x']
286        # [1.4124214, 1.4124214]  # sample std deviation, what we want
287        # numpy.std([[1., 3.]])
288        # 1.0  # population std deviation- not what we want
289        n = len(self.collection)
290        if n < 2:
291            return numpy.nan
292        return float(numpy.std(self.collection) * numpy.sqrt(n / (n - 1)))

421    def _emit_right_join_as_left_join(
422        self, join_node, *, using=None, temp_id_source, sql_format_options=None
423    ):
424        assert join_node.node_name == "NaturalJoinNode"
425        assert join_node.jointype == "RIGHT"
426        # convert to left to avoid SQLite not having a right jone
427        join_node_copy_right = copy.copy(join_node)
428        join_node_copy_right.jointype = "LEFT"
429        join_node_copy_right.sources = [join_node.sources[1], join_node.sources[0]]
430        near_sql_right = data_algebra.db_model.DBModel.natural_join_to_near_sql(
431            self,
432            join_node=join_node_copy_right,
433            using=using,
434            temp_id_source=temp_id_source,
435            sql_format_options=sql_format_options,
436            left_is_first=False,
437        )
438        return near_sql_right

440    def _emit_full_join_as_complex(
441        self, join_node, *, using=None, temp_id_source, sql_format_options=None
442    ):
443        # this is an example how to tree-rewrite the operator platform before emitting SQL.
444        assert join_node.node_name == "NaturalJoinNode"
445        assert join_node.jointype == "FULL"
446        assert len(join_node.on_a) > 0  # could special case zero case later
447        if temp_id_source is None:
448            temp_id_source = [0]
449        if using is None:
450            using = set(join_node.column_names)
451        join_columns = join_node.on_a
452        assert join_node.on_a == join_node.on_b  # TODO: relax this
453        left_descr = join_node.sources[0]
454        right_descr = join_node.sources[1]
455        ops_simulate = (
456            # get shared key set
457            left_descr.project({}, group_by=join_columns)
458            .concat_rows(
459                b=right_descr.project({}, group_by=join_columns),
460                id_column=None,
461            )
462            .project({}, group_by=join_columns)
463            # simulate full join with left joins
464            .natural_join(b=left_descr, on=join_columns, jointype="left")
465            .natural_join(b=right_descr, on=join_columns, jointype="left")
466        )
467        assert isinstance(ops_simulate, NaturalJoinNode)
468        simulate_near_sql = self.natural_join_to_near_sql(
469            join_node=ops_simulate,
470            using=using,
471            temp_id_source=temp_id_source,
472            sql_format_options=sql_format_options,
473        )
474        return simulate_near_sql

476    def natural_join_to_near_sql(
477        self,
478        join_node,
479        *,
480        using=None,
481        temp_id_source=None,
482        sql_format_options=None,
483        left_is_first=True
484    ):
485        """
486        Translate a join into SQL, converting right and full joins to replacement code (as SQLite doesn't have these).
487        """
488        if join_node.node_name != "NaturalJoinNode":
489            raise TypeError(
490                "Expected join_node to be a data_algebra.data_ops.NaturalJoinNode)"
491            )
492        assert left_is_first
493        if temp_id_source is None:
494            temp_id_source = [0]
495        if join_node.jointype == "RIGHT":
496            return self._emit_right_join_as_left_join(
497                join_node,
498                using=using,
499                temp_id_source=temp_id_source,
500                sql_format_options=sql_format_options,
501            )
502        if join_node.jointype == "FULL":
503            return self._emit_full_join_as_complex(
504                join_node,
505                using=using,
506                temp_id_source=temp_id_source,
507                sql_format_options=sql_format_options,
508            )
509        # delegate back to parent class
510        return data_algebra.db_model.DBModel.natural_join_to_near_sql(
511            self,
512            join_node=join_node,
513            using=using,
514            temp_id_source=temp_id_source,
515            sql_format_options=sql_format_options,
516            left_is_first=left_is_first,
517        )

442        self, join_node, *, using=None, temp_id_source, sql_format_options=None
443    ):
444        # this is an example how to tree-rewrite the operator platform before emitting SQL.
445        assert join_node.node_name == "NaturalJoinNode"
446        assert join_node.jointype == "FULL"
447        assert len(join_node.on_a) > 0  # could special case zero case later
448        if temp_id_source is None:
449            temp_id_source = [0]
450        if using is None:
451            using = set(join_node.column_names)
452        join_columns = join_node.on_a
453        assert join_node.on_a == join_node.on_b  # TODO: relax this
454        left_descr = join_node.sources[0]
455        right_descr = join_node.sources[1]
456        ops_simulate = (
457            # get shared key set
458            left_descr.project({}, group_by=join_columns)
459            .concat_rows(
460                b=right_descr.project({}, group_by=join_columns),
461                id_column=None,
462            )
463            .project({}, group_by=join_columns)
464            # simulate full join with left joins
465            .natural_join(b=left_descr, on=join_columns, jointype="left")
466            .natural_join(b=right_descr, on=join_columns, jointype="left")
467        )
468        assert isinstance(ops_simulate, NaturalJoinNode)
469        simulate_near_sql = self.natural_join_to_near_sql(
470            join_node=ops_simulate,
471            using=using,
472            temp_id_source=temp_id_source,
473            sql_format_options=sql_format_options,
474        )
