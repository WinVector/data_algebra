 19class PolarsTerm:
 20    """
 21    Class to carry Polars expression term and annotations about expression tree.
 22    """
 23
 24    polars_term: Any
 25    is_literal: bool
 26    is_column: bool
 27    collect_required: bool = False  # property of tree, not node
 28    one_constant_required: bool  # property of tree, not node
 29
 30    def __init__(
 31        self, 
 32        *, 
 33        polars_term = None, 
 34        is_literal: bool = False,
 35        is_column: bool = False,
 36        collect_required: bool = False,  # property of tree, not node
 37        one_constant_required: bool = False,  # property of tree, not node
 38        inputs: Optional[List] = None,
 39        ) -> None:
 40        """
 41        Carry a Polars expression term (polars_term) plus annotations.
 42
 43        :param polars_term: Optional Polars expression (None means collect info, not a true term)
 44        :param is_literal: True if term is a constant
 45        :param is_column: True if term is a column name
 46        :param collect_required: True if Polars frame collection required by this node or an input node
 47        :param one_constant_required: True one constant required by this node or an input node
 48        :param inputs: inputs to expression node
 49        """
 50        assert isinstance(is_literal, bool)
 51        assert isinstance(is_column, bool)
 52        assert isinstance(collect_required, bool)
 53        assert isinstance(one_constant_required, bool)
 54        assert (is_literal + is_column + (inputs is not None) + (polars_term is None)) == 1
 55        self.polars_term = polars_term
 56        self.is_literal = is_literal
 57        self.collect_required = collect_required
 58        self.one_constant_required = one_constant_required
 59        if inputs is not None:
 60            assert isinstance(inputs, List)
 61            for v in inputs:
 62                self.observe(v)
 63    
 64    def observe(self, v) -> None:
 65        """
 66        Or conditions of v into our conditions
 67        """
 68        assert isinstance(v, PolarsTerm)
 69        if v.collect_required:
 70            self.collect_required = True
 71        if v.one_constant_required:
 72            self.one_constant_required = True

312        """
313        Implementation of Polars evaluation of data algebra operators
314
315        :param op: ViewRepresentation to evaluate
316        :param data_map: dictionary mapping table and view names to data frames

342    def _concat_rows_step(self, op: data_algebra.data_ops_types.OperatorPlatform, *, data_map: Dict[str, Any]):
343        """
344        Execute a concat rows step, returning a data frame.
345        """
346        if op.node_name != "ConcatRowsNode":
347            raise TypeError(
348                "op was supposed to be a data_algebra.data_ops.ConcatRowsNode"
349            )
350        inputs = [self._compose_polars_ops(s, data_map=data_map) for s in op.sources]
351        assert len(inputs) == 2
352        if op.id_column is not None:
353            inputs[0] = inputs[0].with_column(pl.lit(op.a_name).alias(op.id_column))
354            inputs[1] = inputs[1].with_column(pl.lit(op.b_name).alias(op.id_column))
355        res = pl.concat(inputs, how="vertical")
356        return res

364    def _extend_step(self, op: data_algebra.data_ops_types.OperatorPlatform, *, data_map: Dict[str, Any]):
365        """
366        Execute an extend step, returning a data frame.
367        """
368        if op.node_name != "ExtendNode":
369            raise TypeError("op was supposed to be a data_algebra.data_ops.ExtendNode")
370        res = self._compose_polars_ops(op.sources[0], data_map=data_map)
371        partition_by = op.partition_by
372        conditions_from_expressions = PolarsTerm()
373        temp_v_columns = []
374        # see if we need to make partition non-empty
375        if len(partition_by) <= 0:
376            v_name = f"_da_extend_temp_partition_column"
377            partition_by = [v_name]
378            temp_v_columns.append(pl.lit(1).alias(v_name))
379        produced_columns = []
380        for k, opk in op.ops.items():
381            if op.windowed_situation:
382                # enforce is a simple v.f() expression
383                assert isinstance(opk, data_algebra.expr_rep.Expression)
384                if len(opk.args) == 0:
385                    pass
386                elif len(opk.args) == 1:
387                    assert isinstance(opk.args[0], (data_algebra.expr_rep.Value, data_algebra.expr_rep.ColumnReference))
388                    if isinstance(opk.args[0], data_algebra.expr_rep.Value):
389                        # promote value to column for uniformity of API
390                        v_name = f"_da_extend_temp_v_column_{len(temp_v_columns)}"
391                        temp_v_columns.append(pl.lit(opk.args[0].value).alias(v_name))
392                        opk = data_algebra.expr_rep.Expression(
393                            op=opk.op, 
394                            args=[data_algebra.expr_rep.ColumnReference(column_name=v_name)], 
395                            params=opk.params, 
396                            inline=opk.inline, 
397                            method=opk.method,
398                        )
399                else:
400                    raise ValueError(f"can't take {len(opk.args)} arity argument in windowed extend")
401            fld_k_container = opk.act_on(res, data_model=self)  # PolarsTerm
402            assert isinstance(fld_k_container, PolarsTerm)
403            conditions_from_expressions.observe(fld_k_container)
404            fld_k = fld_k_container.polars_term
405            if op.windowed_situation:
406                fld_k = fld_k.over(partition_by)
407            produced_columns.append(fld_k.alias(k))
408        if conditions_from_expressions.one_constant_required:
409            temp_v_columns.append(_da_temp_one_column)
410        assert not conditions_from_expressions.collect_required  # implement if needed
411        if len(temp_v_columns) > 0:
412            res = res.with_columns(temp_v_columns)
413        if len(op.order_by) > 0:
414            order_cols = list(partition_by)
415            partition_set = set(partition_by)
416            for c in op.order_by:
417                if c not in partition_set:
418                    order_cols.append(c)
419            reversed_cols = [True if ci in set(op.reverse) else False for ci in op.order_by]
420            res = res.sort(by=op.order_by, reverse=reversed_cols)
421        res = res.with_columns(produced_columns)  
422        if len(temp_v_columns) > 0:
423            res = res.select(op.columns_produced())
424        return res

426    def _project_step(self, op: data_algebra.data_ops_types.OperatorPlatform, *, data_map: Dict[str, Any]):
427        """
428        Execute a project step, returning a data frame.
429        """
430        if op.node_name != "ProjectNode":
431            raise TypeError("op was supposed to be a data_algebra.data_ops.ProjectNode")
432        res = self._compose_polars_ops(op.sources[0], data_map=data_map)
433        group_by = op.group_by
434        conditions_from_expressions = PolarsTerm()
435        temp_v_columns = []
436        # see if we need to make group_by non-empty
437        if len(group_by) <= 0:
438            v_name = f"_da_project_temp_group_by_column"
439            group_by = [v_name]
440            temp_v_columns.append(pl.lit(1).alias(v_name))
441        produced_columns = []
442        for k, opk in op.ops.items():
443            # enforce is a simple v.f() expression
444            assert isinstance(opk, data_algebra.expr_rep.Expression)
445            if len(opk.args) == 0:
446                pass
447            elif len(opk.args) == 1:
448                assert isinstance(opk.args[0], (data_algebra.expr_rep.Value, data_algebra.expr_rep.ColumnReference))
449                if isinstance(opk.args[0], data_algebra.expr_rep.Value):
450                    # promote value to column for uniformity of API
451                    v_name = f"_da_project_temp_v_column_{len(temp_v_columns)}"
452                    temp_v_columns.append(pl.lit(opk.args[0].value).alias(v_name))
453                    opk = data_algebra.expr_rep.Expression(
454                        op=opk.op, 
455                        args=[data_algebra.expr_rep.ColumnReference(column_name=v_name)], 
456                        params=opk.params, 
457                        inline=opk.inline, 
458                        method=opk.method,
459                    )
460            else:
461                raise ValueError(f"can't take {len(opk.args)} arity argument in project")
462            fld_k_container = opk.act_on(res, data_model=self)  # PolarsTerm
463            assert isinstance(fld_k_container, PolarsTerm)
464            conditions_from_expressions.observe(fld_k_container)
465            fld_k = fld_k_container.polars_term
466            produced_columns.append(fld_k.alias(k))
467        if conditions_from_expressions.one_constant_required:
468            temp_v_columns.append(_da_temp_one_column)
469        assert not conditions_from_expressions.collect_required  # implement if needed
470        if len(temp_v_columns) > 0:
471            res = res.with_columns(temp_v_columns)
472        res = res.groupby(group_by).agg(produced_columns)
473        if len(temp_v_columns) > 0:
474            res = res.select(op.columns_produced())
475        return res

477    def _natural_join_step(self, op: data_algebra.data_ops_types.OperatorPlatform, *, data_map: Dict[str, Any]):
478        """
479        Execute a natural join step, returning a data frame.
480        """
481        if op.node_name != "NaturalJoinNode":
482            raise TypeError(
483                "op was supposed to be a data_algebra.data_ops.NaturalJoinNode"
484            )
485        inputs = [self._compose_polars_ops(s, data_map=data_map) for s in op.sources]
486        assert len(inputs) == 2
487        res = inputs[0].join(
488            inputs[1],
489            left_on=op.on_a,
490            right_on=op.on_b,
491            how=op.jointype.lower(),
492            suffix = "_da_right_tmp",
493        )
494        coalesce_columns = set(op.sources[0].columns_produced()).intersection(op.sources[1].columns_produced()) - set(op.on_a)
495        if len(coalesce_columns) > 0:
496            res = res.with_columns([
497                pl.when(pl.col(c).is_null())
498                    .then(pl.col(c + "_da_right_tmp"))
499                    .otherwise(pl.col(c))
500                    .alias(c)
501                for c in coalesce_columns
502            ])
503            res = res.select(op.columns_produced())
504        return res
