170def default_data_model() -> DataModel:
171    """Get the default (Pandas) data model"""
172    global data_model_type_map
173    return data_model_type_map["default_data_model"]

175pandas_regexp = re.compile(r".*[^a-zA-Z]pandas[^a-zA-Z].*[^a-zA-z]dataframe[^a-zA-Z].*")
176assert pandas_regexp.match("<class 'pandas.core.frame.DataFrame'>".lower()) is not None
177polars_regexp = re.compile(r".*[^a-zA-Z]polars[^a-zA-Z].*[^a-zA-z]dataframe[^a-zA-Z].*")
178assert polars_regexp.match("<class 'polars.internals.dataframe.frame.DataFrame'>".lower()) is not None
179polars_regexp_lazy = re.compile(r".*[^a-zA-Z]polars[^a-zA-Z].*[^a-zA-z]lazyframe[^a-zA-Z].*")
180assert polars_regexp_lazy.match("<class 'polars.internals.lazyframe.frame.LazyFrame'>".lower()) is not None
181
182def lookup_data_model_for_key(key: str) -> DataModel:
183    global data_model_type_map
184    assert isinstance(key, str)
185    key_lower = key.lower()
186    if (key == "default_data_model") or (key == "default_Pandas_model") or (pandas_regexp.match(key_lower) is not None):
187        import data_algebra.pandas_model  # delayed import
188        data_algebra.pandas_model.register_pandas_model(key)
189    elif (key == "default_Polars_model") or (polars_regexp.match(key_lower) is not None) or (polars_regexp_lazy.match(key_lower) is not None):
190        import data_algebra.polars_model  # delayed import
191        data_algebra.polars_model.register_polars_model(key)
192    return data_model_type_map[key]
193
194def lookup_data_model_for_dataframe(d) -> DataModel:
195    key = str(type(d))
196    res = lookup_data_model_for_key(key)
197    assert res.is_appropriate_data_instance(d)
198    return res
