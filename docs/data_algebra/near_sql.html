 14class SQLWithList:
 15    """
 16    Carry an ordered sequence of SQL steps for use with a SQL WITH statement.
 17    """
 18
 19    last_step: "NearSQL"
 20    previous_steps: List[Tuple[str, "NearSQLContainer"]]
 21
 22    def __init__(
 23        self,
 24        *,
 25        last_step: "NearSQL",
 26        previous_steps: Iterable[Tuple[str, "NearSQLContainer"]],
 27    ):
 28        assert isinstance(last_step, NearSQL)
 29        previous_steps = list(previous_steps)
 30        assert isinstance(previous_steps, List)
 31        for vi in previous_steps:
 32            assert isinstance(vi, tuple)
 33            assert len(vi) == 2
 34            assert isinstance(vi[0], str)
 35            assert isinstance(vi[1], NearSQLContainer)
 36        # check no table like entities in previous steps
 37        assert not any([v.near_sql.is_table for k, v in previous_steps])
 38        # check keying is unique
 39        previous_keys = {k for k, v in previous_steps}
 40        if len(previous_keys) != len(previous_steps):
 41            assert len(previous_keys) == len(
 42                previous_steps
 43            )  # so we can set a breakpoint
 44        # check we are not coming up on a duplication
 45        if not last_step.is_table:
 46            assert last_step.quoted_query_name not in previous_keys
 47        self.last_step = last_step
 48        self.previous_steps = previous_steps

123class NearSQLContainer:
124    """
125    NearSQL with bound in columns, force_sql
126    """
127
128    near_sql: NearSQL
129    force_sql: bool
130    columns: Optional[data_algebra.OrderedSet.OrderedSet]
131    public_name: Optional[str]
132    public_name_quoted: Optional[str]
133
134    def __init__(
135        self,
136        *,
137        near_sql: NearSQL,
138        columns: Optional[Iterable[str]] = None,
139        force_sql: bool = False,
140        public_name: Optional[str] = None,
141        public_name_quoted: Optional[str] = None,
142    ):
143        assert isinstance(near_sql, NearSQL)
144        assert isinstance(force_sql, bool)
145        self.near_sql = near_sql
146        self.columns = None
147        if columns is not None:
148            assert not isinstance(columns, str)
149            self.columns = data_algebra.OrderedSet.OrderedSet(columns)
150        self.force_sql = force_sql
151        assert (public_name is not None) == (public_name_quoted is not None)
152        if (public_name is not None) or (public_name_quoted is not None):
153            assert isinstance(public_name, str)
154            assert isinstance(public_name_quoted, str)
155        self.public_name = public_name
156        self.public_name_quoted = public_name_quoted

158    def convert_subsql(
159        self,
160        *,
161        db_model,
162        sql_format_options=None,
163        quoted_query_name_annotation: Optional[str] = None,
164    ) -> List[str]:
165        """Convert subsql, possibly adding query name"""
166        non_trivial_annotation = (quoted_query_name_annotation is not None) and (
167            quoted_query_name_annotation != self.near_sql.quoted_query_name
168        )
169        if isinstance(self.near_sql, data_algebra.near_sql.NearSQLNamedEntity) and (
170            not self.force_sql
171        ):
172            # short circuit table and cte path
173            if non_trivial_annotation:
174                assert quoted_query_name_annotation is not None  # type hint
175                assert isinstance(quoted_query_name_annotation, str)  # type hint
176                sql = [
177                    self.near_sql.quoted_query_name + " " + quoted_query_name_annotation
178                ]  # table name and alias
179            else:
180                sql = [self.near_sql.quoted_query_name]
181        else:
182            # main path
183            sql = self.near_sql.to_sql_str_list(
184                columns=self.columns,
185                force_sql=self.force_sql or non_trivial_annotation,
186                db_model=db_model,
187                sql_format_options=sql_format_options,
188            )
189            if quoted_query_name_annotation is not None:
190                sql = ["("] + sql + [") " + quoted_query_name_annotation]
191        return sql

193    # sequence: a list where last element is a NearSQLContainer previous elements are (name, NearSQLContainer) pairs
194    # stub the replacement common table expression in a NearSQLContainer
195    def to_with_form_stub(
196        self, *, cte_cache: Optional[Dict]
197    ) -> Tuple["NearSQLContainer", List[Tuple[str, "NearSQLContainer"]]]:
198        if self.near_sql.is_table:  # table or common table expression
199            return self, []
200        in_with_form = self.near_sql.to_with_form(cte_cache=cte_cache)
201        sequence = in_with_form.previous_steps
202        stub = in_with_form.last_step
203        assert isinstance(stub, NearSQL)
204        assert isinstance(sequence, List)
205        for v in sequence:
206            assert isinstance(v, tuple)
207            assert len(v) == 2
208            assert isinstance(v[0], str)
209            assert isinstance(v[1], NearSQLContainer)
210        if not stub.is_table:
211            # first check cache
212            ops_key = f"{self.near_sql.ops_key}"
213            if self.columns is not None:
214                ops_key = f"{ops_key}_{list(self.columns)}"
215            if (cte_cache is not None) and (ops_key is not None):
216                try:
217                    retrieved_cte = cte_cache[ops_key]
218                    # copy in context fields
219                    new_stub = NearSQLContainer(
220                        near_sql=retrieved_cte,
221                        columns=self.columns,
222                        force_sql=self.force_sql,
223                        public_name=self.public_name,
224                        public_name_quoted=self.public_name_quoted,
225                    )
226                    return new_stub, []
227                except KeyError:
228                    pass
229            # replace step with a reference
230            if stub.quoted_query_name not in {k for k, v in sequence}:
231                sequence.append(
232                    (
233                        stub.quoted_query_name,
234                        NearSQLContainer(
235                            near_sql=stub,
236                            force_sql=self.force_sql,
237                            columns=self.columns,
238                        ),
239                    )
240                )
241            new_stub_cte = NearSQLCommonTableExpression(
242                query_name=stub.query_name,
243                quoted_query_name=stub.quoted_query_name,
244                ops_key=ops_key,
245            )
246            new_stub = NearSQLContainer(
247                near_sql=new_stub_cte,
248                force_sql=self.force_sql,
249                columns=self.columns,
250                public_name=self.public_name,
251                public_name_quoted=self.public_name_quoted,
252            )
253            if (cte_cache is not None) and (ops_key is not None):
254                cte_cache[ops_key] = new_stub_cte
255        else:
256            assert len(sequence) == 0
257            new_stub = NearSQLContainer(
258                near_sql=stub,
259                force_sql=True,
260                columns=self.columns,
261                public_name=self.public_name,
262                public_name_quoted=self.public_name_quoted,
263            )
264        return new_stub, sequence

483    def to_with_form(self, *, cte_cache: Optional[Dict]) -> SQLWithList:
484        if (
485            self.sub_sql1.near_sql.is_table and self.sub_sql2.near_sql.is_table
486        ):  # TODO: remove this?
487            # tables references don't need to be re-encoded
488            return SQLWithList(last_step=self, previous_steps=[])

499        for stepi in sequence2:
500            nmi = stepi[0]
501            if nmi not in seen:
502                seen.add(nmi)
503                sequence.append(stepi)
504        stubbed_step = NearSQLBinaryStep(
505            terms=self.terms,
506            query_name=self.query_name,
507            quoted_query_name=self.quoted_query_name,
508            sub_sql1=stub1,
509            joiner=self.joiner,
510            sub_sql2=stub2,
511            suffix=self.suffix,
512            annotation=self.annotation,
513            ops_key=self.ops_key,
514        )
515        return SQLWithList(last_step=stubbed_step, previous_steps=sequence)
