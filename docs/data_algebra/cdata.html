 13class RecordSpecification:
 14    """
 15    Class to represent a multi-row data record.
 16    """

 18    def __init__(
 19        self,
 20        control_table,
 21        *,
 22        record_keys=None,
 23        control_table_keys=None,
 24        strict=False,
 25        local_data_model=None
 26    ):
 27        """
 28        :param control_table: data.frame describing record layout
 29        :param record_keys: array of record key column names
 30               defaults to no columns.
 31        :param control_table_keys: array of control_table key column names,
 32               defaults to first column for non-trivial blocks and no columns for rows.
 33        :param strict: logical, if True more checks on transform
 34        :param local_data_model: data.frame data model
 35        """
 36        if local_data_model is None:
 37            local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(control_table)
 38        assert isinstance(local_data_model, data_algebra.data_model.DataModel)
 39        assert local_data_model.is_appropriate_data_instance(control_table)
 40        control_table = local_data_model.clean_copy(control_table)
 41        if control_table.shape[0] < 1:
 42            raise ValueError("control table should have at least 1 row")
 43        if len(control_table.columns) != len(set(control_table.columns)):
 44            raise ValueError("control table columns should be unique")
 45        self.control_table = control_table
 46        assert self.control_table.shape[0] > 0
 47        if record_keys is None:
 48            record_keys = []
 49        if isinstance(record_keys, str):
 50            record_keys = [record_keys]
 51        self.record_keys = [k for k in record_keys]
 52        if control_table_keys is None:
 53            if self.control_table.shape[0] > 1:
 54                control_table_keys = [self.control_table.columns[0]]
 55            else:
 56                control_table_keys = []  # single row records don't need to be keyed
 57        if isinstance(control_table_keys, str):
 58            control_table_keys = [control_table_keys]
 59        if strict and (self.control_table.shape[0] > 1):
 60            if len(control_table_keys) <= 0:
 61                raise ValueError(
 62                    "multi-row records must have at least one control table key"
 63                )
 64        self.control_table_keys = [k for k in control_table_keys]
 65        unknown = set(self.control_table_keys) - set(control_table.columns)
 66        if len(unknown) > 0:
 67            raise ValueError(
 68                "control table keys that are not in the control table: " + str(unknown)
 69            )
 70        if len(self.control_table_keys) >= control_table.shape[1]:
 71            raise ValueError("control table columns must not all be keys")
 72        confused = set(record_keys).intersection(control_table_keys)
 73        if len(confused) > 0:
 74            raise ValueError(
 75                "columns common to record_keys and control_table_keys: " + str(confused)
 76            )
 77        for ck in self.control_table_keys:
 78            if any(local_data_model.bad_column_positions(control_table[ck])):
 79                raise ValueError("NA/NaN/inf/None not allowed as control table keys")
 80        if strict:
 81            if not data_algebra.util.table_is_keyed_by_columns(
 82                self.control_table, self.control_table_keys
 83            ):
 84                raise ValueError("control table wasn't keyed by control table keys")
 85        self.block_columns = self.record_keys + [c for c in self.control_table.columns]
 86        cvs = []
 87        for c in self.control_table.columns:
 88            if c not in self.control_table_keys:
 89                col = self.control_table[c]
 90                isnull = local_data_model.bad_column_positions(col)
 91                if all(isnull):
 92                    raise ValueError("column " + c + " was all null")
 93                for i in range(len(col)):
 94                    if not isnull[i]:
 95                        v = col[i]
 96                        if v not in cvs:
 97                            cvs.append(v)
 98        confused = set(record_keys).intersection(cvs)
 99        if len(confused) > 0:
100            raise ValueError(
101                "control table entries confused with row keys or control table keys"
102            )
103        if strict:
104            if len(set(cvs)) != len(cvs):
105                raise ValueError("duplicate content keys")
106        self.content_keys = cvs
107        self.row_columns = self.record_keys + cvs

185    Class for specifying general record to record transforms.

188    def __init__(
189        self,
190        *,
191        blocks_in: Optional[RecordSpecification] = None,
192        blocks_out: Optional[RecordSpecification] = None
193    ):
194        """
195        Build the transform specification. At least one of blocks_in or blocks_out must not be None.
196
197        :param blocks_in: incoming record specification, None for row-records.
198        :param blocks_out: outgoing record specification, None for row-records.
199        """
200        if blocks_in is not None:
201            assert isinstance(blocks_in, RecordSpecification)
202            ck = [k for k in blocks_in.content_keys if k is not None]
203            if len(ck) != len(set(ck)):
204                raise ValueError("blocks_in can not have duplicate content keys")
205        if blocks_out is not None:
206            assert isinstance(blocks_out, RecordSpecification)
207        if (blocks_in is None) and (blocks_out is None):
208            raise ValueError(
209                "At least one of blocks_in or blocks_out should not be None"
210            )
211        if (blocks_in is not None) and (blocks_out is not None):
212            unknown = set(blocks_out.record_keys) - set(blocks_in.record_keys)
213            if len(unknown) > 0:
214                raise ValueError("unknown outgoing record_keys:" + str(unknown))
215            unknown = set(blocks_out.content_keys) - set(blocks_in.content_keys)
216            if len(unknown) > 0:
217                raise ValueError("unknown outgoing content_keys" + str(unknown))
218        self.blocks_in = blocks_in
219        self.blocks_out = blocks_out
220        if self.blocks_in is not None:
221            self.columns_needed = self.blocks_in.block_columns
222        else:
223            assert self.blocks_out is not None  # type hint
224            self.columns_needed = self.blocks_out.row_columns
225        if self.blocks_out is not None:
226            self.columns_produced = self.blocks_out.block_columns
227        else:
228            assert self.blocks_in is not None  # type hint
229            self.columns_produced = self.blocks_in.row_columns
230        self.fmt_string = self.fmt()
231
232    def __eq__(self, other):
233        if not isinstance(other, RecordMap):
234            return False
235        if (self.blocks_in is None) != (other.blocks_in is None):
236            return False
237        if (self.blocks_out is None) != (other.blocks_out is None):
238            return False
239        if self.blocks_in is not None:
240            if self.blocks_in != other.blocks_in:
241                return False
242        if self.blocks_in is not None:
243            if self.blocks_out != other.blocks_out:
244                return False
245        return True

247    def record_keys(self):
248        """Return keys specifying which set of rows are in a record."""
249        if self.blocks_in is not None:
250            return self.blocks_in.record_keys.copy()
251        if self.blocks_out is not None:
252            return self.blocks_out.record_keys.copy()
253        return None

255    def example_input(self, *, local_data_model=None):
256        """
257        Return example output record.
258
259        :param local_data_model: optional Pandas data model.
260        :return: example result data frame.
261        """
262        
263        
264        if self.blocks_in is not None:
265            example = self.blocks_in.control_table.copy()
266            if local_data_model is None:
267                local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(self.blocks_in.control_table)
268            assert isinstance(local_data_model, data_algebra.data_model.DataModel)
269            nrow = example.shape[0]
270            for rk in self.blocks_in.record_keys:
271                example[rk] = [rk] * nrow
272            return example
273        if self.blocks_out is not None:
274            if local_data_model is None:
275                local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(self.blocks_out.control_table)
276            assert isinstance(local_data_model, data_algebra.data_model.DataModel)
277            example = local_data_model.data_frame()
278            for k in self.blocks_out.row_columns:
279                example[k] = [k]
280            return example
281        return None

283    # noinspection PyPep8Naming
284    def transform(
285        self, X, *, check_blocks_out_keying: bool = False, local_data_model=None
286    ):
287        """
288        Transform X records.
289
290        :param X: data frame to be transformed.
291        :param check_blocks_out_keying: logical, if True check output key constraints.
292        :param local_data_model: pandas data model.
293        :return: transformed data frame.
294        """
295        unknown = set(self.columns_needed) - set(X.columns)
296        if len(unknown) > 0:
297            raise ValueError("missing required columns: " + str(unknown))
298        if local_data_model is None:
299            local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(X)
300        assert isinstance(local_data_model, data_algebra.data_model.DataModel)
301        assert local_data_model.is_appropriate_data_instance(X)
302        X = local_data_model.clean_copy(X)
303        if self.blocks_in is not None:
304            X = local_data_model.blocks_to_rowrecs(
305                X, blocks_in=self.blocks_in
306            )
307        if self.blocks_out is not None:
308            X = local_data_model.rowrecs_to_blocks(
309                X,
310                blocks_out=self.blocks_out,
311                check_blocks_out_keying=check_blocks_out_keying,
312            )
313        return X

315    def compose(self, other):
316        """
317        Experimental method to compose transforms
318        (self.compose(other)).transform(data) == self.transform(other.transform(data))
319
320        :param other: another data_algebra.cdata.RecordMap
321        :return:
322        """
323
324        assert isinstance(other, RecordMap)
325        # (s2.compose(s1)).transform(data) == s2.transform(s1.transform(data))
326        s1 = other
327        s2 = self
328        rk = s1.record_keys()
329        if set(rk) != set(s2.record_keys()):
330            raise ValueError("can only compose operations with matching record_keys")
331        inp = s1.example_input()
332        out = s2.transform(s1.transform(inp))
333        rsi = inp.drop(rk, axis=1, inplace=False)
334        rso = out.drop(rk, axis=1, inplace=False)
335        if inp.shape[0] < 2:
336            if out.shape[0] < 2:
337                return None
338            else:
339                return RecordMap(
340                    blocks_out=RecordSpecification(
341                        control_table=rso,
342                        record_keys=rk,
343                        control_table_keys=s2.blocks_out.control_table_keys,
344                    )
345                )
346        else:
347            if out.shape[0] < 2:
348                return RecordMap(
349                    blocks_in=RecordSpecification(
350                        control_table=rsi,
351                        record_keys=rk,
352                        control_table_keys=s1.blocks_in.control_table_keys,
353                    )
354                )
355            else:
356                return RecordMap(
357                    blocks_in=RecordSpecification(
358                        control_table=rsi,
359                        record_keys=rk,
360                        control_table_keys=s1.blocks_in.control_table_keys,
361                    ),
362                    blocks_out=RecordSpecification(
363                        control_table=rso,
364                        record_keys=rk,
365                        control_table_keys=s2.blocks_out.control_table_keys,
366                    ),
367                )

461def pivot_blocks_to_rowrecs(
462    *,
463    attribute_key_column,
464    attribute_value_column,
465    record_keys,
466    record_value_columns,
467    local_data_model=None
468):
469    """
470    Build a block records to row records map. This is very similar to a SQL pivot.
471
472    :param attribute_key_column: column to identify record attribute keys
473    :param attribute_value_column: column for record attribute values
474    :param record_keys: names of key columns identifying row record blocks
475    :param record_value_columns: names of columns to take row record values from
476    :param local_data_model: data.frame data model
477    :return: RecordMap
478    """
479
480    if local_data_model is None:
481        local_data_model = data_algebra.data_model.default_data_model()
482    assert isinstance(local_data_model, data_algebra.data_model.DataModel)
483    control_table = local_data_model.data_frame(
484        {
485            attribute_key_column: record_value_columns,
486            attribute_value_column: record_value_columns,
487        }
488    )
489    ct = RecordSpecification(
490        control_table,
491        record_keys=record_keys,
492        control_table_keys=[attribute_key_column],
493        local_data_model=local_data_model,
494    )
495    return ct.map_to_rows()

535def pivot_specification(
536    *,
537    row_keys: Iterable[str],
538    col_name_key: str = "column_name",
539    col_value_key: str = "column_value",
540    value_cols: Iterable[str],
541    local_data_model = None,
542) -> RecordMap:
543    """
544    Specify the cdata transformation that pivots records from a single column of values into collected rows.
545    https://en.wikipedia.org/wiki/Pivot_table#History
546
547    :param row_keys: columns that identify rows in the incoming data set
548    :param col_name_key: column name to take the names of columns as a column
549    :param col_value_key: column name to take the values in columns as a column
550    :param value_cols: columns to place values in
551    :param local_data_model: data.frame data model
552    :return: RecordSpecification
553    """
554    assert not isinstance(value_cols, str)
555    value_cols = list(value_cols)
556    assert not isinstance(row_keys, str)
557    row_keys = list(row_keys)
558    assert isinstance(col_name_key, str)
559    assert isinstance(col_value_key, str)
560    known_cols = row_keys + [col_name_key, col_value_key] + value_cols
561    assert len(known_cols) == len(set(known_cols))
562    if local_data_model is None:
563        local_data_model = data_algebra.data_model.default_data_model()
564    assert isinstance(local_data_model, data_algebra.data_model.DataModel)
565    record_map = RecordMap(
566        blocks_in=RecordSpecification(
567            control_table=local_data_model.pd.DataFrame(
568                {
569                    col_name_key: value_cols,
570                    col_value_key: value_cols,
571                }
572            ),
573            record_keys=row_keys,
574            control_table_keys=[col_name_key],
575        )
576    )
577    return record_map
