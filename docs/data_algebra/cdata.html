 20def _format_table(d, *, record_id_cols: Iterable[str], control_id_cols: Iterable[str], add_style: bool = True):
 21    local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(d)
 22    d = local_data_model.to_pandas(d)
 23    pd = data_algebra.data_model.lookup_data_model_for_dataframe(d).pd
 24    record_id_cols = list(record_id_cols)
 25    control_id_cols = list(control_id_cols)

 31    })
 32    if add_style:
 33        d =  (
 34            d.style

 36                .set_properties(**{'background-color': '#7FFFD4'}, subset=control_id_col_pairs)

 41class RecordSpecification:
 42    """
 43    Class to represent a multi-row data record.
 44    """
 45
 46    row_columns: List[str]  # columns when in row form
 47    block_columns: List[str]  # columns when in block form
 48    strict: bool  # do or do not allow repeated value names
 49
 50    def __init__(
 51        self,
 52        control_table,
 53        *,
 54        record_keys=None,
 55        control_table_keys=None,
 56        strict: bool = True,
 57        local_data_model=None,
 58    ):
 59        """
 60        :param control_table: data.frame describing record layout
 61        :param record_keys: array of record key column names
 62               defaults to no columns.
 63        :param control_table_keys: array of control_table key column names,
 64               defaults to first column for non-trivial blocks and no columns for rows.
 65        :param strict: if True don't allow duplicate value names
 66        :param local_data_model: data.frame data model
 67        """
 68        assert isinstance(strict, bool)
 69        self.strict = strict
 70        if local_data_model is None:
 71            local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(control_table)
 72        assert isinstance(local_data_model, data_algebra.data_model.DataModel)
 73        assert local_data_model.is_appropriate_data_instance(control_table)
 74        control_table = local_data_model.clean_copy(control_table)
 75        if control_table.shape[0] < 1:
 76            raise ValueError("control table should have at least 1 row")
 77        if len(control_table.columns) != len(set(control_table.columns)):
 78            raise ValueError("control table columns should be unique")
 79        if strict:
 80            if control_table.shape[0] > 1:
 81                assert control_table_keys is not None
 82                assert len(control_table_keys) > 0
 83                assert local_data_model.table_is_keyed_by_columns(control_table, column_names=control_table_keys)
 84        self.control_table = control_table
 85        assert self.control_table.shape[0] > 0
 86        if record_keys is None:
 87            record_keys = []
 88        if isinstance(record_keys, str):
 89            record_keys = [record_keys]
 90        self.record_keys = list(record_keys)
 91        if control_table_keys is None:
 92            if self.control_table.shape[0] > 1:
 93                control_table_keys = [self.control_table.columns[0]]
 94            else:
 95                control_table_keys = []  # single row records don't need to be keyed
 96        if isinstance(control_table_keys, str):
 97            control_table_keys = [control_table_keys]
 98        if self.control_table.shape[0] > 1:
 99            if len(control_table_keys) <= 0:
100                raise ValueError(
101                    "multi-row records must have at least one control table key"
102                )
103        self.control_table_keys = [k for k in control_table_keys]
104        unknown = set(self.control_table_keys) - set(control_table.columns)
105        if len(unknown) > 0:
106            raise ValueError(
107                "control table keys that are not in the control table: " + str(unknown)
108            )
109        if len(self.control_table_keys) >= control_table.shape[1]:
110            raise ValueError("control table columns must not all be keys")
111        confused = set(record_keys).intersection(control_table_keys)
112        if len(confused) > 0:
113            raise ValueError(
114                "columns common to record_keys and control_table_keys: " + str(confused)
115            )
116        for ck in self.control_table_keys:
117            if any(local_data_model.bad_column_positions(control_table[ck])):
118                raise ValueError("NA/NaN/inf/None not allowed as control table keys")
119        if not local_data_model.table_is_keyed_by_columns(
120            self.control_table, column_names=self.control_table_keys
121        ):
122            raise ValueError("control table wasn't keyed by control table keys")
123        self.block_columns = self.record_keys + [c for c in self.control_table.columns]
124        cvs = []
125        for c in self.control_table.columns:
126            if c not in self.control_table_keys:
127                col = self.control_table[c]
128                isnull = local_data_model.bad_column_positions(col)
129                if any(isnull):
130                    raise ValueError("column " + c + " has null(s)")
131                for i in range(len(col)):
132                    v = col[i]
133                    assert isinstance(v, str)
134                    assert len(v) > 0
135                    cvs.append(v)
136        confused = set(record_keys).intersection(cvs)
137        if len(confused) > 0:
138            raise ValueError(
139                "control table entries confused with row keys or control table keys"
140            )
141        if len(set(cvs)) != len(cvs):
142            if strict:
143                raise ValueError("duplicate content keys")
144            cvs_orig = cvs
145            cvs = []
146            cvs_seen = set()
147            for v in cvs_orig:
148                if v not in cvs_seen:
149                    cvs.append(v)
150                    cvs_seen.add(v)
151        self.content_keys = cvs
152        # columns in the row-record form
153        self.row_columns = self.record_keys + cvs
154        # columns in the block-record form
155        self.block_columns = self.record_keys + list(self.control_table.columns)

261class RecordMap(ShiftPipeAction):
262    """
263    Class for specifying general record to record transforms.
264    """
265
266    blocks_in: Optional[RecordSpecification]
267    blocks_out: Optional[RecordSpecification]
268    strict: bool
269
270    def __init__(
271        self,
272        *,
273        blocks_in: Optional[RecordSpecification] = None,
274        blocks_out: Optional[RecordSpecification] = None,
275        strict: bool = True,
276    ):
277        """
278        Build the transform specification. At least one of blocks_in or blocks_out must not be None.
279
280        :param blocks_in: incoming record specification, None for row-records.
281        :param blocks_out: outgoing record specification, None for row-records.
282        :param strict: if True insist block be strict, and in and out blocks agree on row-form columns.∂
283        """
284        ShiftPipeAction.__init__(self)
285        assert isinstance(strict, bool)
286        self.strict = strict
287        if blocks_in is not None:
288            assert isinstance(blocks_in, RecordSpecification)
289            if strict:
290                assert blocks_in.strict
291            ck = [k for k in blocks_in.content_keys if k is not None]
292            if len(ck) != len(set(ck)):
293                raise ValueError("blocks_in can not have duplicate content keys")
294        if blocks_out is not None:
295            assert isinstance(blocks_out, RecordSpecification)
296            if strict:
297                assert blocks_out.strict
298        if (blocks_in is None) and (blocks_out is None):
299            raise ValueError(
300                "At least one of blocks_in or blocks_out should not be None"
301            )
302        if (blocks_in is not None) and (blocks_out is not None):
303            unknown = set(blocks_out.record_keys) - set(blocks_in.record_keys)
304            if len(unknown) > 0:
305                raise ValueError("unknown outgoing record_keys:" + str(unknown))
306            unknown = set(blocks_out.content_keys) - set(blocks_in.content_keys)
307            if len(unknown) > 0:
308                raise ValueError("unknown outgoing content_keys" + str(unknown))
309            if strict:
310                assert set(blocks_out.record_keys) == set(blocks_in.record_keys)
311        self.blocks_in = blocks_in
312        self.blocks_out = blocks_out
313        if self.blocks_in is not None:
314            self.columns_needed = self.blocks_in.block_columns
315        else:
316            assert self.blocks_out is not None  # type hint
317            self.columns_needed = self.blocks_out.row_columns
318        if self.blocks_out is not None:
319            self.columns_produced = self.blocks_out.block_columns
320        else:
321            assert self.blocks_in is not None  # type hint
322            self.columns_produced = self.blocks_in.row_columns

361    def example_input(
362        self, 
363        *, 
364        local_data_model=None, 
365        value_suffix: str = " value",
366        record_key_suffix: str = " record key",
367        ):
368        """
369        Return example output record.
370
371        :param local_data_model: optional Pandas data model.
372        :param value_suffix: suffix to identify values
373        :param record_key_suffix: suffix to identify record keys
374        :return: example result data frame.
375        """
376        assert isinstance(value_suffix, str)
377        assert isinstance(record_key_suffix, str)
378        rk = self.record_keys()
379        if self.blocks_in is not None:
380            if local_data_model is None:
381                local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(self.blocks_in.control_table)
382            assert isinstance(local_data_model, data_algebra.data_model.DataModel)
383            example = local_data_model.clean_copy(self.blocks_in.control_table)
384            cols = list(example.columns)
385            for j in range(example.shape[1]):
386                if (self.blocks_in.control_table_keys is None) or (cols[j] not in self.blocks_in.control_table_keys):
387                    renames = [f"{local_data_model.get_cell(d=example, row=i, colname=cols[j])}{value_suffix}" for i in range(example.shape[0])]
388                    example = local_data_model.set_col(d=example, colname=cols[j], values=renames)
389        elif self.blocks_out is not None:
390            if local_data_model is None:
391                local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(self.blocks_out.control_table)
392            assert isinstance(local_data_model, data_algebra.data_model.DataModel)
393            example = local_data_model.data_frame({k : [f"{k}{value_suffix}"] for k in self.blocks_out.row_columns if k not in rk})
394        else:
395            raise ValueError("null transform")
396        if len(rk) > 0:
397            nrow = example.shape[0]
398            row_key_frame = local_data_model.data_frame({k: [f"{k}{record_key_suffix}"] * nrow for k in rk})
399            example = local_data_model.concat_columns([row_key_frame, example])
400        return example

403    def transform(
404        self, X, *, local_data_model=None
405    ):
406        """
407        Transform X records.
408
409        :param X: data frame to be transformed.
410        :param local_data_model: pandas data model.
411        :return: transformed data frame.
412        """
413        unknown = set(self.columns_needed) - set(X.columns)
414        if len(unknown) > 0:
415            raise ValueError("missing required columns: " + str(unknown))
416        if local_data_model is None:
417            local_data_model = data_algebra.data_model.lookup_data_model_for_dataframe(X)
418        assert isinstance(local_data_model, data_algebra.data_model.DataModel)
419        assert local_data_model.is_appropriate_data_instance(X)
420        X = local_data_model.clean_copy(X)
421        if self.blocks_in is not None:
422            X = local_data_model.blocks_to_rowrecs(
423                X, blocks_in=self.blocks_in
424            )
425        if self.blocks_out is not None:
426            X = local_data_model.rowrecs_to_blocks(
427                X,
428                blocks_out=self.blocks_out,
429            )
430        return X

432    def compose(self, other):
433        """
434        Experimental method to compose transforms
435        (self.compose(other)).transform(data) == self.transform(other.transform(data))
436
437        :param other: another data_algebra.cdata.RecordMap
438        :return:
439        """
440
441        assert isinstance(other, RecordMap)
442        # (s2.compose(s1)).transform(data) == s2.transform(s1.transform(data))
443        s1 = other
444        s2 = self
445        rk = s1.record_keys()
446        if set(rk) != set(s2.record_keys()):
447            raise ValueError("can only compose operations with matching record_keys")
448        inp = s1.example_input()
449        out = s2.transform(s1.transform(inp))
450        rsi = inp.drop(rk, axis=1, inplace=False)
451        rso = out.drop(rk, axis=1, inplace=False)
452        strict = self.strict and other.strict
453        if inp.shape[0] < 2:
454            if out.shape[0] < 2:
455                return None
456            else:
457                return RecordMap(
458                    blocks_out=RecordSpecification(
459                        control_table=rso,
460                        record_keys=rk,
461                        control_table_keys=s2.blocks_out.control_table_keys,
462                        strict=strict,
463                    ),
464                    strict=strict,
465                )
466        else:
467            if out.shape[0] < 2:
468                return RecordMap(
469                    blocks_in=RecordSpecification(
470                        control_table=rsi,
471                        record_keys=rk,
472                        control_table_keys=s1.blocks_in.control_table_keys,
473                        strict=strict,
474                    ),
475                    strict=strict,
476                )
477            else:
478                return RecordMap(
479                    blocks_in=RecordSpecification(
480                        control_table=rsi,
481                        record_keys=rk,
482                        control_table_keys=s1.blocks_in.control_table_keys,
483                        strict=strict,
484                    ),
485                    blocks_out=RecordSpecification(
486                        control_table=rso,
487                        record_keys=rk,
488                        control_table_keys=s2.blocks_out.control_table_keys,
489                        strict=strict,
490                    ),
491                    strict=strict,
492                )

494    def as_pipeline(self, 
495        *, 
496        table_name: Optional[str] = None,
497        local_data_model=None,
498        value_suffix: str = " value",
499        record_key_suffix: str = " record key",

501        """
502        Build into processing pipeline.
503
504        :param table_name: name for input table.
505        :param local_data_model: optional Pandas data model.
506        :param value_suffix: suffix to identify values
507        :param record_key_suffix: suffix to identify record keys

519                table_name=table_name,
520                row_limit=None,
521                keep_sample=True,
522                keep_all=True,
523                )
524                .convert_records(self)

536    def act_on(self, b, *, correct_ordered_first_call: bool = False):
537        assert isinstance(correct_ordered_first_call, bool)
538        if isinstance(b, RecordMap):
539            self.compose(b)
540        if isinstance(b, data_algebra.data_ops.ViewRepresentation):
541            return b.convert_records(self)
542        if correct_ordered_first_call and isinstance(b, ShiftPipeAction):
543            return b.act_on(self, correct_ordered_first_call=False)  # fall back to peer's action
544        # assume table like
545        return self.transform(b)
